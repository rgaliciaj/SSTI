CREATE VIEW DB_SERVICIOTI.V_TICKET AS
SELECT T.CODIGO_TICKET        AS CODIGO_TICKET,
       U.USER_ID              AS CODIGO_USUARIO,
       T.FECHA_CREACION       AS FECHA_CREACION,
       T.DESCRIPCION          AS DESCRIPCION,
       E.NOMBRE_ESTADO        AS ESTADO_TICKET,
       P.NOMBRE_PRIORIDAD     AS CODIGO_PRIORIDAD,
       T.FECHA_MODIFICACION   AS FECHA_MODIFICACION,
       UU.USER_ID             AS CODIGO_TECNICO,
       RI.DESCRIPCION_RECURSO AS CODIGO_RECURSO,
       RU.NOMBRE_RUTA         AS RUTA
FROM TICKET T
         LEFT JOIN USUARIOS U ON T.CODIGO_USUARIO = U.CODIGO_USUARIO
         LEFT JOIN ESTADOS E ON T.ESTADO_TICKET = E.CODIGO_ESTADO
         LEFT JOIN PRIORIDAD P ON T.CODIGO_PRIORIDAD = P.CODIGO_PRIORIDAD
         LEFT JOIN RECURSOS_INFORMATICOS RI ON T.CODIGO_RECURSO = RI.CODIGO_RECURSO
         LEFT JOIN RUTA RU ON T.RUTA = RU.CODIGO_RUTA
         LEFT JOIN USUARIOS UU ON UU.CODIGO_USUARIO = T.CODIGO_TECNICO
/
